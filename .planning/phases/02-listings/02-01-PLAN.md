  ```python
  import httpx
  from app.config import settings

  class OllamaClient:
      def __init__(self) -> None:
          self.base_url = settings.ollama_base_url
          self.model = settings.ollama_model
          self.embed_model = settings.ollama_embed_model

      async def health(self) -> bool:
          """Return True if Ollama server is reachable. 2s timeout."""
          try:
              async with httpx.AsyncClient(timeout=2.0) as client:
                  r = await client.get(f"{self.base_url}/api/tags")
                  return r.status_code == 200
          except Exception:
              return False

      async def generate(self, prompt: str, system: str = "") -> str:
          """Non-streaming generation. Returns empty string on failure."""
          payload: dict = {"model": self.model, "prompt": prompt, "stream": False}
          if system:
              payload["system"] = system
          async with httpx.AsyncClient(timeout=60.0) as client:
              r = await client.post(f"{self.base_url}/api/generate", json=payload)
              r.raise_for_status()
              return r.json().get("response", "")

      async def embed(self, text: str) -> list[float]:
          """Generate 768-dim embedding vector. Returns empty list on failure."""
          async with httpx.AsyncClient(timeout=15.0) as client:
              r = await client.post(
                  f"{self.base_url}/api/embeddings",
                  json={"model": self.embed_model, "prompt": text},
              )
              r.raise_for_status()
              return r.json().get("embedding", [])

      async def generate_stream(self, prompt: str, system: str = ""):
          """Async generator yielding token strings."""
          payload: dict = {"model": self.model, "prompt": prompt, "stream": True}
          if system:
              payload["system"] = system
          async with httpx.AsyncClient(timeout=120.0) as client:
              async with client.stream("POST", f"{self.base_url}/api/generate", json=payload) as response:
                  response.raise_for_status()
                  async for line in response.aiter_lines():
                      if line.strip():
                          import json
                          try:
                              data = json.loads(line)
                              token = data.get("response", "")
                              if token:
                                  yield token
                              if data.get("done"):
                                  break
                          except json.JSONDecodeError:
                              continue

  ollama = OllamaClient()
  ```

  - Price anomaly (weight 0.3): `_price_anomaly(listing)`
  - Owner reputation (weight 0.2): `_owner_reputation(owner_id)`
  - LLM consistency (weight 0.5): `_llm_consistency(listing)`
  - Returns `round(min(1.0, max(0.0, total)), 3)`

  **`_price_anomaly(listing: dict) -> float`**:
  - Requires category, city, price. Returns 0.0 if any missing.
  - Fetch up to 100 active listings with same category+city from DB.
  - Compute avg_price from fetched prices.
  - If no data → 0.0. If 0.3 <= price/avg <= 3.0 → 0.0. If ratio < 0.1 or > 10.0 → 1.0. Otherwise → 0.5.
  - Catch all DB exceptions → return 0.0.

  **`_owner_reputation(owner_id: str) -> float`**:
  - Count owner's rejected listings.
  - 0 rejected → 0.0. 1-2 → 0.3. 3-5 → 0.6. 6+ → 1.0.
  - Catch all exceptions → 0.0.

  **`_llm_consistency(listing: dict) -> float`**:

  - If no description → return 0.0.
  - Build system prompt: fraud detection for real estate, look for unrealistic claims, off-platform payment requests, suspicious urgency. Return JSON `{"fraud_score": 0.0-1.0, "reason": "..."}`.
  - Build user prompt with listing attributes (title, price, category, property_type, city, bedrooms, bathrooms, size_sqm) + description.
  - Parse JSON from response (find first `{` to last `}`).
  - Catch all exceptions → return 0.0.

  Create `backend/app/uploads/router.py`:

  ```python
  import uuid
  from fastapi import APIRouter, HTTPException, Depends
  from pydantic import BaseModel
  from app.database import supabase_admin
  from app.dependencies import get_current_user

  router = APIRouter()

  ALLOWED_BUCKETS = {"avatars", "listing-images", "attachments"}

  class SignedUrlRequest(BaseModel):
      bucket: str
      filename: str

  @router.post("/signed-url")
  async def get_signed_upload_url(
      body: SignedUrlRequest,
      current_user: dict = Depends(get_current_user),
  ):
      if body.bucket not in ALLOWED_BUCKETS:
          raise HTTPException(status_code=400, detail=f"Invalid bucket. Allowed: {', '.join(ALLOWED_BUCKETS)}")

      user_id = current_user["id"]
      ext = body.filename.rsplit(".", 1)[-1] if "." in body.filename else ""
      unique_name = f"{uuid.uuid4().hex}.{ext}" if ext else uuid.uuid4().hex
      storage_path = f"{user_id}/{unique_name}"

      try:
          response = supabase_admin.storage.from_(body.bucket).create_signed_upload_url(storage_path)
      except Exception as e:
          raise HTTPException(status_code=500, detail=f"Failed to generate signed URL: {e}")

      if isinstance(response, dict):
          signed_url = response.get("signedURL") or response.get("signed_url") or ""
          path = response.get("path") or storage_path
      else:
          signed_url = getattr(response, "signed_url", "") or getattr(response, "signedURL", "")
          path = getattr(response, "path", storage_path)

      public_url = f"{supabase_admin.storage.from_(body.bucket).get_public_url(path)}"
      return {"upload_url": signed_url, "path": path, "public_url": public_url, "bucket": body.bucket}
  ```
