3. Create backend/app/middleware/rate_limit.py:
   ```python
   from slowapi import Limiter
   from slowapi.util import get_remote_address
   try:
       from slowapi.wrappers import RedisStorage
       from redis.asyncio import from_url as redis_from_url
   except ImportError:
       RedisStorage = None

   def create_limiter(redis_url: str | None) -> Limiter:
       storage = None
       if redis_url and RedisStorage:
           try:
               client = redis_from_url(redis_url, decode_responses=True)
               storage = RedisStorage(client)
           except Exception:
               pass  # fail-open: use in-memory
       return Limiter(key_func=get_remote_address, storage_uri=None if storage is None else redis_url)
   ```
   Export `limiter = create_limiter(settings.redis_url)` at module level.

1. Create backend/app/middleware/security_headers.py as a Starlette BaseHTTPMiddleware:
   ```python
   from starlette.middleware.base import BaseHTTPMiddleware
   from starlette.requests import Request

   SECURITY_HEADERS = {
       "X-Content-Type-Options": "nosniff",
       "X-Frame-Options": "DENY",
       "X-XSS-Protection": "1; mode=block",
       "Strict-Transport-Security": "max-age=31536000; includeSubDomains",
       "Referrer-Policy": "strict-origin-when-cross-origin",
       "Permissions-Policy": "geolocation=(), microphone=(), camera=()",
   }

   class SecurityHeadersMiddleware(BaseHTTPMiddleware):
       async def dispatch(self, request: Request, call_next):
           response = await call_next(request)
           for header, value in SECURITY_HEADERS.items():
               response.headers[header] = value
           return response
   ```
