```python
"""Security regression tests — IDOR, injection, auth bypass."""
import pytest
from httpx import AsyncClient
from app.main import app
from app.dependencies import get_current_user

FAKE_USER_A_ID = "00000000-0000-0000-0000-000000000001"
FAKE_USER_B_ID = "00000000-0000-0000-0000-000000000002"

# Minimal user objects matching what get_current_user returns
fake_user_a = {"id": FAKE_USER_A_ID, "role": "user", "email": "usera@test.com"}
fake_user_b = {"id": FAKE_USER_B_ID, "role": "user", "email": "userb@test.com"}


@pytest.mark.asyncio
async def test_idor_listing_update(client: AsyncClient, test_listing_owned_by_user_a):
    """User B cannot update a listing owned by User A."""
    # Override get_current_user to return user B
    app.dependency_overrides[get_current_user] = lambda: fake_user_b
    listing_id = test_listing_owned_by_user_a["id"]
    resp = await client.put(
        f"/api/listings/{listing_id}",
        json={"title": "Malicious overwrite"},
    )
    app.dependency_overrides.clear()
    assert resp.status_code == 403, f"Expected 403, got {resp.status_code}"


@pytest.mark.asyncio
async def test_idor_message_read(client: AsyncClient, test_conversation_owned_by_user_a):
    """User B cannot read a conversation they are not party to."""
    app.dependency_overrides[get_current_user] = lambda: fake_user_b
    convo_id = test_conversation_owned_by_user_a["id"]
    resp = await client.get(f"/api/messages/conversations/{convo_id}")
    app.dependency_overrides.clear()
    assert resp.status_code in (403, 404), f"Expected 403/404, got {resp.status_code}"


@pytest.mark.asyncio
async def test_sql_injection_search(client: AsyncClient):
    """SQL injection in search query must never cause a 500."""
    malicious = "'; DROP TABLE listings; --"
    resp = await client.get("/api/listings/search", params={"q": malicious})
    assert resp.status_code != 500, f"SQL injection caused 500: {resp.text}"
    assert resp.status_code in (200, 400, 422)


@pytest.mark.asyncio
async def test_admin_endpoint_no_token(client: AsyncClient):
    """Admin endpoints must return 401 without a valid token."""
    resp = await client.get("/api/admin/users")
    assert resp.status_code == 401, f"Expected 401, got {resp.status_code}"
```

import http from "k6/http";
import { sleep } from "k6";
import { check } from "k6";

export const options = {
  stages: [
    { duration: "10s", target: 20 },
    { duration: "20s", target: 100 },
    { duration: "10s", target: 0 },
  ],
  thresholds: {
    // CI exits non-zero if p95 exceeds 500ms
    "http_req_duration": ["p(95)<500"],
    // At least 95% of requests must succeed
    "http_req_failed": ["rate<0.05"],
  },
};

const BASE_URL = __ENV.BASE_URL || "http://localhost:8000";

export default function () {
  const res = http.get(`${BASE_URL}/api/listings`, {
    headers: { "Accept": "application/json" },
  });

  check(res, {
    "status is 200 or 401": (r) => r.status === 200 || r.status === 401,
    "response time < 1000ms": (r) => r.timings.duration < 1000,
  });

  sleep(0.5);
}
