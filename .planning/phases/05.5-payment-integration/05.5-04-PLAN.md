<!-- Paymob HMAC-SHA512 valid signature construction for tests -->
```python
import hashlib, hmac

def make_valid_paymob_hmac(obj: dict, secret: str) -> str:
    FIELDS = ["amount_cents","created_at","currency","error_occured","has_parent_transaction",
              "id","integration_id","is_3d_secure","is_auth","is_capture","is_refunded",
              "is_standalone_payment","is_voided","order.id","owner","pending",
              "source_data.pan","source_data.sub_type","source_data.type","success"]
    def get_nested(obj, key):
        parts = key.split(".")
        val = obj
        for p in parts:
            val = val.get(p) if isinstance(val, dict) else None
        return str(val) if val is not None else ""
    concatenated = "".join(get_nested(obj, f) for f in FIELDS)
    return hmac.new(secret.encode(), concatenated.encode(), hashlib.sha512).hexdigest()
```

    merchant_ref = "test-merchant-ref-003"
    transaction_obj = _make_paymob_transaction(merchant_ref, success=True)

    mock_admin.table.return_value.select.return_value.eq.return_value.maybe_single.return_value.execute.return_value.data = {
        "id": "pay-003",
        "status": "paid",  # already processed

        resp = client.post(
            "/api/payments/paymob/callback",
            json={"obj": transaction_obj, "hmac": "any-value-hmac-is-mocked"},

    assert resp.json()["status"] == "already_processed"
