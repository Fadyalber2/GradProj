```tsx
import { useQuery } from "@tanstack/react-query";
```

```tsx
"use client";

import { useState } from "react";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { paymentQueries } from "@/lib/queries";
import { api } from "@/lib/api";
import type { ApiSubscriptionPlan } from "@/types/api";

interface AgencySubscriptionCardProps {
  agencyId: string;
  currentPlan?: "none" | "starter" | "pro" | "enterprise";
}

export function AgencySubscriptionCard({ agencyId, currentPlan = "none" }: AgencySubscriptionCardProps) {
  const [selectedMethod, setSelectedMethod] = useState<"paymob" | "fawry">("paymob");

  const { data: plansData, isLoading } = useQuery(paymentQueries.subscriptionPlans());

  const plans: ApiSubscriptionPlan[] = plansData?.plans ?? [];

  const handleSubscribe = async (planId: string) => {
    await api.post("/api/payments/subscriptions", {
      agency_id: agencyId,
      plan: planId,
      payment_method: selectedMethod,
    });
  };

  if (isLoading) {
    return <div className="animate-pulse h-48 rounded-lg bg-muted" />;
  }

  return (
    <div className="space-y-4">
      <h3 className="text-lg font-semibold">Agency Subscription Plans</h3>
      <div className="grid gap-4 sm:grid-cols-3">
        {plans.map((plan) => (
          <div
            key={plan.id}
            className={`rounded-lg border p-4 flex flex-col gap-3 ${
              currentPlan === plan.id ? "border-primary bg-primary/5" : "border-border"
            }`}
          >
            <div>
              <p className="font-semibold">{plan.name}</p>
              <p className="text-2xl font-bold">
                EGP {plan.price_egp.toLocaleString()}
                <span className="text-sm font-normal text-muted-foreground">/mo</span>
              </p>
            </div>
            <p className="text-sm text-muted-foreground">
              Up to {plan.listing_quota === 999 ? "unlimited" : plan.listing_quota} listings
            </p>
            {currentPlan === plan.id ? (
              <span className="text-xs text-primary font-medium">Current plan</span>
            ) : (
              <button
                onClick={() => handleSubscribe(plan.id)}
                className="mt-auto w-full rounded-md bg-primary py-2 text-sm font-medium text-primary-foreground hover:bg-primary/90 transition-colors"
              >
                Subscribe
              </button>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```

```tsx
"use client";

import { useState } from "react";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { api } from "@/lib/api";
import { PaymentMethodSelector } from "@/components/payments/PaymentMethodSelector";
import { PaymobCheckout } from "@/components/payments/PaymobCheckout";
import { FawryCheckout } from "@/components/payments/FawryCheckout";
import type { ApiDeal, ApiPaymobInitiateResponse, ApiFawryInitiateResponse } from "@/types/api";

const dealSchema = z.object({
  listing_id: z.string().uuid("Must be a valid listing ID"),
  deal_type: z.enum(["sale", "rental"]),
  deal_value_egp: z.number().positive("Deal value must be positive"),
});

type DealFormValues = z.infer<typeof dealSchema>;

interface DealReportModalProps {
  onClose: () => void;
}

export function DealReportModal({ onClose }: DealReportModalProps) {
  const [step, setStep] = useState<"form" | "pay" | "paymob-checkout" | "fawry-checkout">("form");
  const [paymentMethod, setPaymentMethod] = useState<"paymob" | "fawry">("paymob");
  const [deal, setDeal] = useState<ApiDeal | null>(null);
  const [paymobData, setPaymobData] = useState<ApiPaymobInitiateResponse | null>(null);
  const [fawryData, setFawryData] = useState<ApiFawryInitiateResponse | null>(null);

  const queryClient = useQueryClient();

  const { register, handleSubmit, formState: { errors, isSubmitting } } = useForm<DealFormValues>({
    resolver: zodResolver(dealSchema),
    defaultValues: { deal_type: "sale" },
  });

  const onSubmit = async (values: DealFormValues) => {
    const created = await api.post<ApiDeal>("/api/deals", values);
    setDeal(created);
    setStep("pay");
  };

  const handleInitiatePayment = async () => {
    if (!deal) return;
    if (paymentMethod === "paymob") {
      const data = await api.post<ApiPaymobInitiateResponse>("/api/payments/initiate", { deal_id: deal.id });
      setPaymobData(data);
      setStep("paymob-checkout");
    } else {
      const data = await api.post<ApiFawryInitiateResponse>("/api/payments/fawry/initiate", { deal_id: deal.id });
      setFawryData(data);
      setStep("fawry-checkout");
    }
    queryClient.invalidateQueries({ queryKey: ["payment-history"] });
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/60">
      <div className="relative w-full max-w-lg rounded-xl bg-background p-6 shadow-xl">
        <button onClick={onClose} className="absolute right-4 top-4 text-muted-foreground hover:text-foreground" aria-label="Close">✕</button>

        {step === "form" && (
          <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
            <h2 className="text-xl font-semibold">Report a Closed Deal</h2>
            <div className="space-y-2">
              <label className="text-sm font-medium">Listing ID</label>
              <input {...register("listing_id")} className="w-full rounded-md border bg-background px-3 py-2 text-sm" placeholder="Paste listing UUID" />
              {errors.listing_id && <p className="text-xs text-destructive">{errors.listing_id.message}</p>}
            </div>
            <div className="space-y-2">
              <label className="text-sm font-medium">Deal Type</label>
              <select {...register("deal_type")} className="w-full rounded-md border bg-background px-3 py-2 text-sm">
                <option value="sale">Sale</option>
                <option value="rental">Rental</option>
              </select>
            </div>
            <div className="space-y-2">
              <label className="text-sm font-medium">Deal Value (EGP)</label>
              <input
                type="number"
                {...register("deal_value_egp", { valueAsNumber: true })}
                className="w-full rounded-md border bg-background px-3 py-2 text-sm"
                placeholder="e.g. 2500000"
              />
              {errors.deal_value_egp && <p className="text-xs text-destructive">{errors.deal_value_egp.message}</p>}
            </div>
            <button type="submit" disabled={isSubmitting} className="w-full rounded-md bg-primary py-2 text-sm font-medium text-primary-foreground disabled:opacity-50">
              {isSubmitting ? "Computing commission..." : "Calculate Commission"}
            </button>
          </form>
        )}

        {step === "pay" && deal && (
          <div className="space-y-4">
            <h2 className="text-xl font-semibold">Commission Due</h2>
            <div className="rounded-lg bg-muted/50 p-4 space-y-1">
              <div className="flex justify-between text-sm">
                <span className="text-muted-foreground">Deal Value</span>
                <span>EGP {deal.deal_value_egp.toLocaleString()}</span>
              </div>
              <div className="flex justify-between text-sm">
```
