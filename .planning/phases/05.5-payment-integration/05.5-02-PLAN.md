**`backend/app/payments/paymob.py`**: Paymob gateway client.

CRITICAL rules:
- Use `httpx.AsyncClient` ALWAYS — never synchronous `httpx.get()`/`httpx.post()` (would block async event loop)
- Use `logging.getLogger(__name__)` — NEVER `print()`
- Import: `import hashlib, hmac as hmac_mod, logging; from app.config import settings`

Implement these functions:

```python
BASE_URL = "https://accept.paymob.com/api"

async def get_auth_token(api_key: str) -> str:
    """Step 1: Exchange API key for short-lived auth token."""
    async with httpx.AsyncClient() as client:
        resp = await client.post(f"{BASE_URL}/auth/tokens", json={"api_key": api_key})
        resp.raise_for_status()
        return resp.json()["token"]

async def register_order(auth_token: str, amount_cents: int, merchant_order_id: str) -> str:
    """Step 2: Register order. Returns order_id string."""
    async with httpx.AsyncClient() as client:
        resp = await client.post(
            f"{BASE_URL}/ecommerce/orders",
            json={
                "auth_token": auth_token,
                "delivery_needed": False,
                "amount_cents": amount_cents,
                "currency": "EGP",
                "merchant_order_id": merchant_order_id,
                "items": [],
            },
        )
        resp.raise_for_status()
        return str(resp.json()["id"])

async def get_payment_key(
    auth_token: str, order_id: str, amount_cents: int,
    integration_id: int, billing_data: dict
) -> str:
    """Step 3: Obtain payment_key for iframe embedding."""
    async with httpx.AsyncClient() as client:
        resp = await client.post(
            f"{BASE_URL}/acceptance/payment_keys",
            json={
                "auth_token": auth_token,
                "amount_cents": amount_cents,
                "expiration": 3600,
                "order_id": order_id,
                "billing_data": billing_data,
                "currency": "EGP",
                "integration_id": integration_id,
                "lock_order_when_paid": True,
            },
        )
        resp.raise_for_status()
        return resp.json()["token"]

async def initiate_payment(amount_cents: int, merchant_order_id: str, billing_data: dict) -> tuple[str, str]:
    """Orchestrate all 3 Paymob steps. Returns (payment_key, order_id).
    Execute all 3 steps without unrelated awaits between them to avoid auth token expiry (Pitfall 2).
    """
    token = await get_auth_token(settings.paymob_api_key)
    order_id = await register_order(token, amount_cents, merchant_order_id)
    payment_key = await get_payment_key(token, order_id, amount_cents, settings.paymob_integration_id, billing_data)
    return payment_key, order_id
```

HMAC verification — exact 20-field list in this lexicographic order:
`amount_cents`, `created_at`, `currency`, `error_occured`, `has_parent_transaction`, `id`, `integration_id`, `is_3d_secure`, `is_auth`, `is_capture`, `is_refunded`, `is_standalone_payment`, `is_voided`, `order.id`, `owner`, `pending`, `source_data.pan`, `source_data.sub_type`, `source_data.type`, `success`

```python
PAYMOB_HMAC_FIELDS = [
    "amount_cents", "created_at", "currency", "error_occured",
    "has_parent_transaction", "id", "integration_id", "is_3d_secure",
    "is_auth", "is_capture", "is_refunded", "is_standalone_payment",
    "is_voided", "order.id", "owner", "pending",
    "source_data.pan", "source_data.sub_type", "source_data.type", "success",
]

def verify_paymob_hmac(transaction_obj: dict, received_hmac: str, hmac_secret: str) -> bool:
    """Verify Paymob webhook HMAC-SHA512. Uses constant-time compare_digest."""
    def get_nested(obj: dict, key: str) -> str:
        parts = key.split(".")
        val: object = obj
        for part in parts:
            val = val.get(part) if isinstance(val, dict) else None
        return str(val) if val is not None else ""

    concatenated = "".join(get_nested(transaction_obj, f) for f in PAYMOB_HMAC_FIELDS)
    calculated = hmac_mod.new(
        hmac_secret.encode(),
        concatenated.encode(),
        hashlib.sha512,
    ).hexdigest()
    return hmac_mod.compare_digest(calculated, received_hmac)  # constant-time, no timing attack
```

**`backend/app/payments/fawry.py`**: Fawry gateway client.

Define charge and status URLs as separate constants — do NOT derive status URL via string manipulation on the charge URL constant (fragile and not confirmed by Fawry docs).

```python
import hashlib, hmac as hmac_mod, logging, httpx
from app.config import settings

STAGING_CHARGE_URL = "https://atfawry.fawrystaging.com/ECommerceWeb/Fawry/payments/charge"
PRODUCTION_CHARGE_URL = "https://atfawry.com/ECommerceWeb/Fawry/payments/charge"
STAGING_STATUS_URL = "https://atfawry.fawrystaging.com/ECommerceWeb/Fawry/payments/status"
PRODUCTION_STATUS_URL = "https://atfawry.com/ECommerceWeb/Fawry/payments/status"

def _fawry_charge_signature(
    merchant_code: str, merchant_ref: str, customer_profile_id: str,
    payment_method: str, amount: str, secure_key: str
) -> str:
    """SHA-256 of concatenated fields (no separator).
    amount MUST be f"{amount_egp:.2f}" — NOT str(float) which drops trailing zeros (Pitfall 3)."""
    data = merchant_code + merchant_ref + customer_profile_id + payment_method + amount + secure_key
    return hashlib.sha256(data.encode()).hexdigest()

async def create_fawry_charge(
    merchant_ref: str, customer_mobile: str, amount_egp: float, description: str
) -> dict:
    """Returns dict with referenceNumber key (the cash reference code)."""
    amount_str = f"{amount_egp:.2f}"  # CRITICAL: must be exactly "580.50" not "580.5"
    sig = _fawry_charge_signature(
        settings.fawry_merchant_code, merchant_ref, customer_mobile,
        "PayAtFawry", amount_str, settings.fawry_secure_key
    )
    url = STAGING_CHARGE_URL if settings.fawry_sandbox else PRODUCTION_CHARGE_URL
    async with httpx.AsyncClient() as client:
        resp = await client.post(url, json={
            "merchantCode": settings.fawry_merchant_code,
            "merchantRefNum": merchant_ref,
            "customerProfileId": customer_mobile,
            "customerMobile": customer_mobile,
            "paymentMethod": "PayAtFawry",
            "currencyCode": "EGP",
            "amount": amount_str,
            "signature": sig,
            "chargeItems": [{"itemId": merchant_ref, "description": description, "price": amount_str, "quantity": 1}],
        })
        resp.raise_for_status()
        return resp.json()

async def get_fawry_status(merchant_ref: str) -> dict:
    """Poll Fawry for order status. Returns dict with orderStatus field."""
    base_url = STAGING_STATUS_URL if settings.fawry_sandbox else PRODUCTION_STATUS_URL
    url = f"{base_url}?merchantRefNum={merchant_ref}&merchantCode={settings.fawry_merchant_code}"
    async with httpx.AsyncClient() as client:
        resp = await client.get(url)
        resp.raise_for_status()
        return resp.json()

def verify_fawry_callback(params: dict, secure_key: str) -> bool:
    """Verify FawryPay server notification messageSignature (SHA-256).
    Field order: requestId + fawryRefNumber + merchantRefNumber + paymentAmount +
    orderAmount + fawryFees + orderStatus + paymentMethod + secureKey"""
    data = (
        str(params.get("requestId", ""))
        + str(params.get("fawryRefNumber", ""))
        + str(params.get("merchantRefNumber", ""))
        + str(params.get("paymentAmount", ""))
        + str(params.get("orderAmount", ""))
        + str(params.get("fawryFees", ""))
        + str(params.get("orderStatus", ""))
        + str(params.get("paymentMethod", ""))
        + secure_key
    )
    calculated = hashlib.sha256(data.encode()).hexdigest()
    return hmac_mod.compare_digest(calculated, params.get("messageSignature", ""))
```

```python
from datetime import datetime, timezone, timedelta

logger = logging.getLogger(__name__)

SUBSCRIPTION_PLANS = {
    "starter":    50_000,   # 500 EGP in piasters
    "pro":       120_000,   # 1200 EGP
    "enterprise": 250_000,  # 2500 EGP
}
SUBSCRIPTION_LISTING_QUOTAS = {
    "starter": 10,
    "pro": 50,
    "enterprise": 999,
}

def to_piasters(egp: float) -> int:
    """Convert EGP amount to piasters (integer). Uses Decimal to avoid float rounding (Pitfall 1)."""
    return int(Decimal(str(egp)) * 100)

def to_egp(piasters: int) -> float:
    """Convert piasters to EGP."""
    return piasters / 100

def compute_commission(deal_type: str, deal_value_egp: float) -> tuple[float, int]:
    """Returns (commission_pct_display, commission_amount_piasters).
    Uses Decimal arithmetic to avoid float rounding errors (Pitfall 1).
    Example: ('sale', 2500000) -> (2.0, 5000000000)  [2% of 2.5M EGP = 50,000 EGP = 5B piasters]
    """
    rate = settings.commission_rate_sale if deal_type == "sale" else settings.commission_rate_rental
    commission_egp = Decimal(str(deal_value_egp)) * Decimal(str(rate))
    commission_piasters = int(commission_egp.quantize(Decimal("1"), rounding=ROUND_HALF_UP) * 100)
    return rate * 100, commission_piasters  # (2.0, piasters)
```

```python
@payments_router.post("/paymob/callback", status_code=200)
async def paymob_callback(request: Request):
    body = await request.json()
    transaction_obj = body.get("obj", {})
    received_hmac = body.get("hmac", "")

    if not verify_paymob_hmac(transaction_obj, received_hmac, settings.paymob_hmac_secret):
        raise HTTPException(status_code=400, detail="Invalid HMAC")

    if not transaction_obj.get("success", False):
        return {"status": "ignored"}

    merchant_ref = str(transaction_obj.get("order", {}).get("merchant_order_id", ""))

    # Idempotency check — UNIQUE constraint on merchant_ref in DB, but check status here too
    existing = supabase_admin.table("payments").select("id, status").eq("merchant_ref", merchant_ref).maybe_single().execute()
    if existing.data and existing.data["status"] == "paid":
        return {"status": "already_processed"}

    await payment_service.mark_payment_paid(merchant_ref=merchant_ref, provider="paymob")
    return {"status": "ok"}

@payments_router.post("/fawry/callback", status_code=200)
async def fawry_callback(request: Request):
    # Fawry sends x-www-form-urlencoded — parse as form
    form_data = await request.form()
    params = dict(form_data)

    if not verify_fawry_callback(params, settings.fawry_secure_key):
        raise HTTPException(status_code=400, detail="Invalid signature")

    if params.get("orderStatus", "") != "PAID":
        return {"status": "ignored"}

    merchant_ref = params.get("merchantRefNumber", "")
    existing = supabase_admin.table("payments").select("id, status").eq("merchant_ref", merchant_ref).maybe_single().execute()
    if existing.data and existing.data["status"] == "paid":
        return {"status": "already_processed"}

    await payment_service.mark_payment_paid(merchant_ref=merchant_ref, provider="fawry")
    return {"status": "ok"}
```
